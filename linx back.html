const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); const bcrypt = require('bcrypt'); const jwt = require('jsonwebtoken'); const app = express(); app.use(express.json()); app.use(cors()); // MongoDB connect mongoose.connect('mongodb://127.0.0.1:27017/linx'); // Schema const User = mongoose.model('User', { username: String, email: String, password: String }); // SIGNUP app.post('/signup', async (req, res) => { const { username, email, password } = req.body; const hashed = await bcrypt.hash(password, 10); const user = new User({ username, email, password: hashed }); await user.save(); res.json({ message: "User created" }); }); // LOGIN app.post('/login', async (req, res) => { const { email, password } = req.body; const user = await User.findOne({ email }); if (!user) return res.json({ message: "User not found" }); const match = await bcrypt.compare(password, user.password); if (!match) return res.json({ message: "Wrong password" }); const token = jwt.sign({ id: user._id }, "secret"); res.json({ message: "Login success", token }); }); app.listen(5000, () => console.log("Server running on port 5000"));
